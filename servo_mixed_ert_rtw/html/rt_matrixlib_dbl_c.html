14void rt_ComplexTimes_Dbl(creal_T* c,



15  const creal_T a,



16  const creal_T b)



17{



18  if (a.im == 0.0) {



19    c->re = a.re * b.re;



20    c->im = a.re * b.im;



21    if (b.im == 0.0) {



22      c->im = 0.0;



23    } else if (b.re == 0.0 || a.re == 0.0) {



24      c->re = 0.0;



25    }



26  } else if (b.im == 0.0) {



27    c->re = a.re * b.re;



28    c->im = a.im * b.re;



29    if (b.re == 0.0 || a.re == 0.0) {



30      c->re = 0.0;



31    }



32  } else if (a.re == 0.0) {



33    c->re = -a.im * b.im;



34    c->im = a.im * b.re;



35    if (b.re == 0.0) {



36      c->im = 0.0;



37    }



38  } else if (b.re == 0.0) {



39    c->re = -a.im * b.im;



40    c->im = a.re * b.im;



41  } else {



42    c->re = a.re * b.re - a.im * b.im;



43    c->im = a.re * b.im + a.im * b.re;



44  }



45}

47void rt_ComplexRDivide_Dbl(creal_T* c,



48  const creal_T a,



49  const creal_T b)



50{



51  if (b.im == 0.0) {



52    c->re = a.re / b.re;



53    if (a.im == 0.0) {



54      c->im = 0.0;



55    } else {                           /* a.im != 0.0 */



56      c->im = a.im / b.re;



57      if (a.re == 0.0) {



58        c->re = 0.0;



59      }



60    }



61  } else if (b.re == 0.0) {            /* b.im != 0.0 */



62    c->re = a.im / b.im;



63    c->im = -a.re / b.im;



64    if (a.re == 0.0) {



65      c->im = 0.0;



66    } else if (a.im == 0.0) {



67      c->re = 0.0;



68    }



69  } else {



70    real_T brm = fabs(b.re);



71    real_T bim = fabs(b.im);



72    if (brm > bim) {



73      real_T s = b.im / b.re;



74      real_T d = b.re + s * b.im;



75      c->im = (a.im - s * a.re) / d;



76      c->re = (a.re + s * a.im) / d;



77    } else if (brm == bim) {



78      real_T half = 0.5;



79      real_T sgnbr = b.re > 0.0 ? half : -half;



80      real_T sgnbi = b.im > 0.0 ? half : -half;



81      c->im = (a.im*sgnbr - a.re*sgnbi)/brm;



82      c->re = (a.re*sgnbr + a.im*sgnbi)/brm;



83    } else {



84      real_T s = b.re / b.im;



85      real_T d = b.im + s * b.re;



86      c->im = (s * a.im - a.re) / d;



87      c->re = (s * a.re + a.im) / d;



88    }



89  }



90}

92void rt_ComplexReciprocal_Dbl(creal_T* Out,



93  const creal_T In1)



94{



95  creal_T y;



96  y.re = 1.0;



97  y.im = 0.0;



98  rt_ComplexRDivide_Dbl(Out, y, In1);



99}

103real_T rt_Hypot_Dbl(real_T a, real_T b)



104{



105  real_T y;



106  if (a != a || b != b) {



107    y = a + b;



108  } else {



109    real_T t;



110    if (fabs(a) > fabs(b)) {



111      t = b/a;



112      y = fabs(a)*sqrt(1.0 + t*t);



113    } else {



114      if (b == 0.0) {



115        y = 0.0;



116      } else {



117        t = a/b;



118        y = fabs(b)*sqrt(1.0 + t*t);



119      }



120    }



121  }



122



123  return y;



124}
